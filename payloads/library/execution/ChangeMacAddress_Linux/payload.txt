ATTACKMODE HID

# Requirements:
#  - Permissions

# VARIABLES
# You need to know the sudo password and replace 'SUDO_PASS' with this
SUDO-PASS='example'
# Set here your preferred MAC, you can don't change it remaining with the default value
NEW_MAC='example'

QUACK DELAY 1500
QUACK CTRL-ALT t
QUACK DELAY 2000

#### PERMISSIONS SECTION ####
QUACK STRING sudo su
QUACK ENTER
QUACK DELAY 1000
QUACK STRING $SUDO-PASS
QUACK ENTER
QUACK DELAY 1000

#### MAC SECTION ####
# net-tools command
QUACK STRING apt install net-tools
QUACK ENTER
QUACK DELAY 2000

# Get the net interface name
QUACK STRING INTERFACE=\$(ip route get 8.8.8.8 # awk '{print \$5}')
QUACK ENTER
QUACK DELAY 500
QUACK STRING ifconfig \$INTERFACE down
QUACK ENTER
QUACK DELAY 500
QUACK STRING ifconfig \$INTERFACE hw ether $NEW_MAC
QUACK ENTER
QUACK DELAY 500
QUACK STRING ifconfig \$INTERFACE up
QUACK ENTER
QUACK DELAY 500


#### REMOVE TRACES ####
QUACK DELAY 2000
QUACK STRING history -c; exit
QUACK ENTER
